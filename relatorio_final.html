\textbf{EP2\_3} \textbf{Classe Impressora} — Encapsulamento com Getters e Setters

Um dos pilares da Programação Orientada a Objetos (POO) é o \textbf{encapsulamento}, que protege os dados internos de um objeto e controla seu acesso por meio de métodos públicos.

Sua tarefa é:

\begin{itemize}[itemsep=2pt, parsep=0pt]
\item Criar a classe \textbf{Impressora} com \textbf{atributos privados}: modelo e folhasRestantes.
\item Utilizar um \textbf{construtor} para inicializar os atributos.
\item Criar métodos públicos:
  \begin{itemize}[itemsep=1pt, parsep=0pt]
    \item \texttt{getModelo()} e \texttt{setModelo()} para acessar e modificar o modelo.
    \item \texttt{getFolhasRestantes()} apenas para leitura das folhas.
    \item \texttt{imprimir( paginas )} que reduz as folhas disponíveis e retorna uma mensagem de status.
    \item \texttt{status\_impressora()} para exibir informações completas.
  \end{itemize}
\item Apresentar a saída formatada para cada método escolhido na entrada de dados, conforme exemplo a seguir.
\end{itemize}

import json
import random

class Impressora:
    def __init__(self, modelo, folhas):
        self.__modelo = modelo
        self.__folhasRestantes = folhas

    def getModelo(self):
        return self.__modelo

    def setModelo(self, novo_modelo):
        self.__modelo = novo_modelo

    def getFolhasRestantes(self):
        return self.__folhasRestantes

    def imprimir(self, paginas):
        if paginas <= 0:
            return "Quantidade de páginas inválida."
        if paginas > self.__folhasRestantes:
            return "Folhas insuficientes."
        self.__folhasRestantes -= paginas
        return f"Imprimindo {paginas} páginas..."

    def status_impressora(self):
        return f"Modelo: {self.__modelo}, Folhas restantes: {self.__folhasRestantes}"

inp_list, out_list = [], []

modelos = ["LaserJet Pro", "DeskJet 2700", "EcoTank L3150", "Pixma G3020", "OfficeJet 8015"]
num_test_cases = 5

for i in range(num_test_cases):
    mod = random.choice(modelos)
    fol = random.randint(15, 50)
    imp = Impressora(mod, fol)

    op = random.choice(["status", "imprimir", "mudar_mod", "folhas"])
    if op == "status":
        entrada = f"{mod}, {fol}; status_impressora()"
        saida = imp.status_impressora()
    elif op == "imprimir":
        pgs = random.randint(5, 20)
        entrada = f"{mod}, {fol}; imprimir({pgs})"
        saida = imp.imprimir(pgs)
    elif op == "mudar_mod":
        novo = random.choice(modelos)
        imp.setModelo(novo)
        entrada = f"{mod}, {fol}; setModelo('{novo}'); status_impressora()"
        saida = imp.status_impressora()
    else:  # folhas
        entrada = f"{mod}, {fol}; getFolhasRestantes()"
        saida = imp.getFolhasRestantes()

    inp_list.append(entrada)
    out_list.append(saida)

cases = {"input": inp_list, "output": out_list}
moodle_cases = json.dumps(cases)
caso0_inp = inp_list[0]
caso0_out = out_list[0]
